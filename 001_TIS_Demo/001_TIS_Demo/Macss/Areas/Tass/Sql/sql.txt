SELECT HINCOD  , HINNAM  , HINNMK  , TORCOD
     , KISYUA  , KISYUB  , DTMOTO  , TKCOD
     , SYUBTU  , CTLFL1  , KHINCD  , '0' AS DELFLG
     , CRTCOD  , CRTYMD  , UPDCOD  , UPDYMD

UNION ALL

     , KISYUA  , KISYUB  , DTMOTO  , TKCOD
     , SYUBTU  , CTLFL1  , KHINCD  , DELFLG
